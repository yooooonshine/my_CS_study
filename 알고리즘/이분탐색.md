	int mid;

		mid = (low + high) / 2; 
		if(key == arr[mid]) { 
		// 탐색 성공

		// 왼쪽 부분 arr[0]부터 arr[mid-1]에서의 탐색 
			return binarySearch1(key ,low, mid-1); 
		} else { 
		// 오른쪽 부분 - arr[mid+1]부터 arr[high]에서의 탐색 
			return binarySearch1(key, mid+1, high); 
		} 
	} 
	return -1; // 탐색 실패
